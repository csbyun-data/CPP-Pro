* Reference Variable
```
// 윤성우의 C++ Chap 2-3
int main(void) {
  int num1 = 100;
  int &num2 = num1;

  cout << "Value: " << num1 << endl;
  cout << "Ref  : " << num2 << endl;
  cout << "Value: " << &num1 << endl;
  cout << "Ref  : " << &num2 << endl;

  return 0;
}
```
